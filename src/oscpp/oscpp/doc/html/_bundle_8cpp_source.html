51

60 uint64_t ntp = timeTag_.toNTP();

67 high = htonl(high);
68 low = htonl(low);

71 const std::byte* highBytes = reinterpret_cast<const std::byte*>(&high);
72 result.insert(result.end(), highBytes, highBytes + sizeof(uint32_t));

150 // Deserialize a bundle from binary data
151 Bundle Bundle::deserialize(const std::byte* data, size_t size) {
152 if (!data) {
153 throw OSCException("Null data pointer provided to deserialize",
154 OSCException::ErrorCode::InvalidArgument);
155 }
156
157 if (size < 16) { // Minimum size: "#bundle\0" (8 bytes with padding) + time tag (8 bytes)
158 throw OSCException("Bundle data too small: " + std::to_string(size) +
159 " bytes (minimum required: 16 bytes)",
160 OSCException::ErrorCode::MalformedPacket);
161 }
162
163 try {
164 // 1. Check for "#bundle" identifier
165 const char* cdata = reinterpret_cast<const char*>(data);
166 if (std::strncmp(cdata, "#bundle", 7) != 0) {
167 throw OSCException("Not an OSC bundle (missing #bundle identifier)",
168 OSCException::ErrorCode::MalformedPacket);
169 }
170
171 // Skip past "#bundle" string (8 bytes with padding)
172 size_t pos = 8;
173
174 // 2. Extract time tag (8 bytes)
175 if (pos + 8 > size) {
176 throw OSCException("Bundle data truncated: missing time tag",
177 OSCException::ErrorCode::MalformedPacket);
178 }
179
180 uint32_t seconds, fraction;
181 std::memcpy(&seconds, data + pos, sizeof(uint32_t));
182 std::memcpy(&fraction, data + pos + sizeof(uint32_t), sizeof(uint32_t));
183
184 // Convert from big-endian
185 seconds = ntohl(seconds);
186 fraction = ntohl(fraction);
187
188 // Create TimeTag from seconds and fraction
189 TimeTag timeTag(seconds, fraction);
190
191 // Create the bundle with extracted time tag
192 Bundle bundle(timeTag);
193
194 // Move past time tag
195 pos += 8;
196
197 // 3. Process bundle elements
198 while (pos + 4 <= size) { // Need at least 4 bytes for element size
199 // Get element size (int32, big-endian)
200 uint32_t elementSizeBE;
201 std::memcpy(&elementSizeBE, data + pos, sizeof(uint32_t));
202 uint32_t elementSize = ntohl(elementSizeBE);
203
204 // Move past size field
205 pos += 4;
206
207 // Validate element size
208 if (elementSize == 0) {
209 throw OSCException("Invalid element size: zero-length element",
210 OSCException::ErrorCode::MalformedPacket);
211 }
212
213 if (pos + elementSize > size) {
214 throw OSCException("Invalid element size or truncated bundle: claimed size " +
215 std::to_string(elementSize) +
216 " bytes exceeds remaining data",
217 OSCException::ErrorCode::MalformedPacket);
218 }
219
220 // Check if this element is a nested bundle or a message
221 if (elementSize >= 8 && std::strncmp(cdata + pos, "#bundle", 7) == 0) {
222 // Nested bundle - recursively deserialize
223 try {
224 Bundle childBundle = Bundle::deserialize(data + pos, elementSize);
225 bundle.addBundle(childBundle);
226 } catch (const OSCException& e) {
227 // Add context to the exception and rethrow
228 throw OSCException("Error in nested bundle: " + std::string(e.what()),
229 e.code());
230 }
231 } else {
232 // Regular message - parse and add to the bundle
233 try {
234 Message message = Message::deserialize(data + pos, elementSize);
235 bundle.addMessage(message);
236 } catch (const OSCException& e) {
237 // Add context to the exception and rethrow
238 throw OSCException("Error in bundled message: " + std::string(e.what()),
239 e.code());
240 }
241 }
242
243 // Move to the next element
244 pos += elementSize;
245 }
246
247 // Ensure we consumed all the data - this is more of a sanity check since
248 // we allow extra bytes at the end for forward compatibility
249 if (pos < size) {
250 // This is just a warning - we don't throw an exception for this case
251 // Some implementations might add padding or extra data
252 }
253
254 return bundle;
255
256 } catch (const OSCException&) {
257 // Re-throw OSC exceptions (already properly formatted)
258 throw;
259 } catch (const std::exception& e) {
260 // Wrap other exceptions with additional context
261 throw OSCException("Error deserializing OSC bundle: " + std::string(e.what()),
262 OSCException::ErrorCode::DeserializationError);
263 } catch (...) {
264 // Catch any other unexpected errors
265 throw OSCException("Unknown error during OSC bundle deserialization",
266 OSCException::ErrorCode::DeserializationError);
