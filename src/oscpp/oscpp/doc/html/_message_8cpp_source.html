165 // Serialize the message to OSC format
166 std::vector<std::byte> Message::serialize() const {
167 // Pre-calculate buffer size to avoid reallocations
168 size_t totalSize = 0;
169
170 // Calculate address size with padding
171 size_t pathLen = path_.length() + 1; // Include null terminator
172 size_t paddedPathSize = padSize(pathLen);
173 totalSize += paddedPathSize;
174
175 // Calculate type tag size with padding
176 std::string typeTag = ",";
177 for (const auto &arg : arguments_) {
178 typeTag += arg.typeTag();
179 }
180 size_t typeTagLen = typeTag.length() + 1;
181 size_t paddedTypeTagSize = padSize(typeTagLen);
182 totalSize += paddedTypeTagSize;
183
184 // Estimate argument sizes (conservatively)
185 for (const auto &arg : arguments_) {
186 char tag = arg.typeTag();
187
188 // Skip array markers as they don't have binary representation
189 if (tag == ARRAY_BEGIN_TAG || tag == ARRAY_END_TAG) {
190 continue;
191 }
192
193 // T, F, N, I types have no binary data
194 if (tag == TRUE_TAG || tag == FALSE_TAG || tag == NIL_TAG || tag == INFINITUM_TAG) {
195 continue;
196 }
197
198 // Add estimated size based on type
199 switch (tag) {
200 case INT32_TAG:
201 case FLOAT_TAG:
202 case CHAR_TAG:
203 case RGBA_TAG:
204 case MIDI_TAG:
205 totalSize += 4;
206 break;
207 case INT64_TAG:
208 case DOUBLE_TAG:
209 case TIMETAG_TAG:
210 totalSize += 8;
211 break;
212 case STRING_TAG:
213 case SYMBOL_TAG:
214 // String length + null terminator + padding
215 if (tag == STRING_TAG) {
216 const auto &str = arg.asString();
217 totalSize += padSize(str.length() + 1);
218 } else {
219 const auto &sym = arg.asSymbol();
220 totalSize += padSize(sym.length() + 1);
221 }
222 break;
223 case BLOB_TAG:
224 // Size field + blob data + padding
225 const auto &blob = arg.asBlob();
226 totalSize += 4 + padSize(blob.size());
227 break;
228 }
229 }
230
231 // Create result vector with pre-allocated size
232 std::vector<std::byte> result;
233 result.reserve(totalSize);
234
235 // 1. Add Address Pattern (null-terminated, padded to 4-byte boundary)
236 const char *pathStr = path_.c_str();
237
238 // Add path string with null terminator
239 const std::byte *pathBytes = reinterpret_cast<const std::byte *>(pathStr);
240 result.insert(result.end(), pathBytes, pathBytes + pathLen);
241
242 // Add padding to align to 4-byte boundary
243 result.resize(paddedPathSize, std::byte{0});
244
245 // 2. Add Type Tag String (starts with ',', null-terminated, padded)
246 // Add type tag string with null terminator
247 const char *typeTagStr = typeTag.c_str();
248 const std::byte *typeTagBytes = reinterpret_cast<const std::byte *>(typeTagStr);
249 result.insert(result.end(), typeTagBytes, typeTagBytes + typeTagLen);
250
251 // Add padding to align to 4-byte boundary
252 result.resize(paddedPathSize + paddedTypeTagSize, std::byte{0});
253
254 // 3. Add Argument Data
255 for (const auto &arg : arguments_) {
256 // Skip array markers as they don't have a binary representation
257 if (arg.typeTag() == ARRAY_BEGIN_TAG || arg.typeTag() == ARRAY_END_TAG) {
258 continue;
259 }
260
261 // T, F, N, I types have no binary data
262 if (arg.typeTag() == TRUE_TAG || arg.typeTag() == FALSE_TAG ||
263 arg.typeTag() == NIL_TAG || arg.typeTag() == INFINITUM_TAG) {
264 continue;
265 }
266
267 // Serialize the argument value
268 std::vector<std::byte> argData;
269 argData.reserve(16); // Small initial reservation for most argument types
270 arg.serialize(argData);
271 result.insert(result.end(), argData.begin(), argData.end());
272 }
273
274 return result;
275 }

277 // Deserialize a message from binary data
278 Message Message::deserialize(const std::byte *data, size_t size) {
279 if (size < 4) {
280 throw OSCException("Message data too small", OSCException::ErrorCode::MalformedPacket);
281 }
282
283 // 1. Extract OSC Address Pattern
284 const char *cdata = reinterpret_cast<const char *>(data);
285 if (cdata[0] != '/') {
286 throw OSCException("Invalid OSC address pattern (must start with '/')",
287 OSCException::ErrorCode::AddressError);
288 }
289
290 // Find the null terminator for the address pattern
291 size_t pathEnd = 0;
292 while (pathEnd < size && cdata[pathEnd] != '\0') {
293 pathEnd++;
294 }
295
296 if (pathEnd >= size) {
297 throw OSCException("Missing null terminator in OSC address pattern",
298 OSCException::ErrorCode::MalformedPacket);
299 }
300
301 // Extract the address pattern string
302 std::string path(cdata, pathEnd);
303
304 // Create message object with validated path
305 Message message(path);
306
307 // Move position past the address pattern including padding
308 size_t pos = padSize(pathEnd + 1); // +1 for null terminator
309
310 if (pos >= size) {
311 // No type tag string or arguments
312 return message;
313 }
314
315 // 2. Extract Type Tag String
316 if (cdata[pos] != ',') {
317 throw OSCException("Type tag string must start with ','",
318 OSCException::ErrorCode::MalformedPacket);
319 }
320
321 // Find the null terminator for the type tag string
322 size_t typeTagStart = pos;
323 size_t typeTagEnd = pos;
324
325 while (typeTagEnd < size && cdata[typeTagEnd] != '\0') {
326 typeTagEnd++;
327 }
328
329 if (typeTagEnd >= size) {
330 throw OSCException("Missing null terminator in type tag string",
331 OSCException::ErrorCode::MalformedPacket);
332 }
333
334 // Extract the type tag string
335 std::string typeTag(cdata + typeTagStart, typeTagEnd - typeTagStart);
336
337 // Move position past the type tag string including padding
338 pos = padSize(typeTagEnd + 1); // +1 for null terminator
339
340 // 3. Extract Arguments based on Type Tags
341 bool inArray = false;
342
343 // Start from index 1 to skip the initial ',' character
344 for (size_t i = 1; i < typeTag.length(); i++) {
345 char tag = typeTag[i];
346
347 // Handle array markers
348 if (tag == Value::ARRAY_BEGIN_TAG) {
349 inArray = true;
350 message.arguments_.push_back(Value::arrayBegin());
351 continue;
352 }
353
354 if (tag == Value::ARRAY_END_TAG) {
355 inArray = false;
356 message.arguments_.push_back(Value::arrayEnd());
357 continue;
358 }
359
360 // Handle other value types using Value::deserialize
361 if (pos <= size) {
362 size_t remainingSize = size - pos;
363 const std::byte *dataPtr = data + pos;
364 // Create the Value directly in the vector to avoid copy construction
365 message.arguments_.push_back(Value::deserialize(dataPtr, remainingSize, tag));
366 pos = size - remainingSize; // Update position based on bytes consumed
367 } else {
368 throw OSCException("Message data truncated while parsing arguments",
369 OSCException::ErrorCode::MalformedPacket);
370 }
371 }
372
373 return message;
374 }
