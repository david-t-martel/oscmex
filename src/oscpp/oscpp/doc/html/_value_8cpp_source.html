41 inline uint64_t bigEndian(uint64_t x) {
42#ifdef _WIN32
43 // Windows doesn't have htonll by default
44 const uint32_t hi = htonl(static_cast<uint32_t>(x >> 32));
45 const uint32_t lo = htonl(static_cast<uint32_t>(x & 0xFFFFFFFF));
46 return (static_cast<uint64_t>(lo) << 32) | hi;
47#else
48 return htobe64(x);
49#endif
50 }

250 // Serialization methods
251 void Value::serialize(std::vector<std::byte>& buffer) const {
252 try {
253 switch (typeTag()) {
254 case INT32_TAG: {
255 Int32 val = bigEndian(asInt32());
256 const std::byte* bytes = reinterpret_cast<const std::byte*>(&val);
257 buffer.insert(buffer.end(), bytes, bytes + sizeof(Int32));
258 break;
259 }
260 case INT64_TAG: {
261 Int64 val = bigEndian(asInt64());
262 const std::byte* bytes = reinterpret_cast<const std::byte*>(&val);
263 buffer.insert(buffer.end(), bytes, bytes + sizeof(Int64));
264 break;
265 }
266 case FLOAT_TAG: {
267 Float val = bigEndian(asFloat());
268 const std::byte* bytes = reinterpret_cast<const std::byte*>(&val);
269 buffer.insert(buffer.end(), bytes, bytes + sizeof(Float));
270 break;
271 }
272 case DOUBLE_TAG: {
273 Double val = bigEndian(asDouble());
274 const std::byte* bytes = reinterpret_cast<const std::byte*>(&val);
275 buffer.insert(buffer.end(), bytes, bytes + sizeof(Double));
276 break;
277 }
278 case STRING_TAG:
279 case SYMBOL_TAG: {
280 const std::string& str = (typeTag() == STRING_TAG) ? asString() : asSymbol();
281 const size_t strSize = str.size() + 1; // Include null terminator
282 const size_t paddedSize = padSize(strSize);
283
284 // Add the string with null terminator
285 const std::byte* bytes = reinterpret_cast<const std::byte*>(str.c_str());
286 buffer.insert(buffer.end(), bytes, bytes + strSize);
287
288 // Add padding
289 for (size_t i = strSize; i < paddedSize; ++i) {
290 buffer.push_back(std::byte{0});
291 }
292 break;
293 }
294 case BLOB_TAG: {
295 const Blob& blob = asBlob();
296 const size_t blobSize = blob.size();
297
298 // Add blob size (big endian)
299 Int32 size = bigEndian(static_cast<Int32>(blobSize));
300 const std::byte* sizeBytes = reinterpret_cast<const std::byte*>(&size);
301 buffer.insert(buffer.end(), sizeBytes, sizeBytes + sizeof(Int32));
302
303 // Add blob data
304 buffer.insert(buffer.end(), blob.bytes(), blob.bytes() + blobSize);
305
306 // Add padding if needed
307 const size_t paddedSize = padSize(blobSize);
308 for (size_t i = blobSize; i < paddedSize; ++i) {
309 buffer.push_back(std::byte{0});
310 }
311 break;
312 }
313 case TIMETAG_TAG: {
314 const TimeTag& tag = asTimeTag();
315 uint64_t ntp = bigEndian(tag.toNTP());
316 const std::byte* bytes = reinterpret_cast<const std::byte*>(&ntp);
317 buffer.insert(buffer.end(), bytes, bytes + sizeof(uint64_t));
318 break;
319 }
320 case CHAR_TAG: {
321 Int32 val = bigEndian(static_cast<Int32>(asChar()));
322 const std::byte* bytes = reinterpret_cast<const std::byte*>(&val);
323 buffer.insert(buffer.end(), bytes, bytes + sizeof(Int32));
324 break;
325 }
326 case RGBA_TAG: {
327 const RGBAColor& color = asRGBA();
328 uint32_t rgba = (static_cast<uint32_t>(color.r) << 24) |
329 (static_cast<uint32_t>(color.g) << 16) |
330 (static_cast<uint32_t>(color.b) << 8) |
331 static_cast<uint32_t>(color.a);
332 rgba = bigEndian(rgba);
333 const std::byte* bytes = reinterpret_cast<const std::byte*>(&rgba);
334 buffer.insert(buffer.end(), bytes, bytes + sizeof(uint32_t));
335 break;
336 }
337 case MIDI_TAG: {
338 const MIDIMessage& midi = asMIDI();
339 buffer.push_back(std::byte{midi.bytes[0]});
340 buffer.push_back(std::byte{midi.bytes[1]});
341 buffer.push_back(std::byte{midi.bytes[2]});
342 buffer.push_back(std::byte{midi.bytes[3]});
343 break;
344 }
345 case TRUE_TAG:
346 case FALSE_TAG:
347 case NIL_TAG:
348 case INFINITUM_TAG:
349 // These tags have no data
350 break;
351 default:
352 throw UnknownTypeException("Cannot serialize unknown type '" +
353 std::string(1, typeTag()) + "'");
354 }
355 } catch (const OSCException&) {
356 // Re-throw OSC exceptions
357 throw;
358 } catch (const std::exception& e) {
359 // Wrap other exceptions
360 throw SerializationException("Error serializing value: " + std::string(e.what()));
361 }
362 }

364 Value Value::deserialize(const std::byte*& data, size_t& remainingSize, char typeTag) {
365 if (!data) {
366 throw InvalidArgumentException("Null data pointer in Value::deserialize");
367 }
368
369 if (remainingSize == 0) {
370 throw BufferOverflowException("No data available for deserializing type '" +
371 std::string(1, typeTag) + "'");
372 }
373
374 try {
375 switch (typeTag) {
376 case INT32_TAG: {
377 if (remainingSize < sizeof(Int32)) {
378 throw BufferOverflowException("Not enough data for Int32");
379 }
380 Int32 val;
381 std::memcpy(&val, data, sizeof(Int32));
382 val = bigEndian(val);
383 data += sizeof(Int32);
384 remainingSize -= sizeof(Int32);
385 return Value(val);
386 }
387 case INT64_TAG: {
388 if (remainingSize < sizeof(Int64)) {
389 throw BufferOverflowException("Not enough data for Int64");
390 }
391 Int64 val;
392 std::memcpy(&val, data, sizeof(Int64));
393 val = bigEndian(val);
394 data += sizeof(Int64);
395 remainingSize -= sizeof(Int64);
396 return Value(val);
397 }
398 case FLOAT_TAG: {
399 if (remainingSize < sizeof(Float)) {
400 throw BufferOverflowException("Not enough data for Float");
401 }
402 Float val;
403 std::memcpy(&val, data, sizeof(Float));
404 val = bigEndian(val);
405 data += sizeof(Float);
406 remainingSize -= sizeof(Float);
407 return Value(val);
408 }
409 case DOUBLE_TAG: {
410 if (remainingSize < sizeof(Double)) {
411 throw BufferOverflowException("Not enough data for Double");
412 }
413 Double val;
414 std::memcpy(&val, data, sizeof(Double));
415 val = bigEndian(val);
416 data += sizeof(Double);
417 remainingSize -= sizeof(Double);
418 return Value(val);
419 }
420 case STRING_TAG:
421 case SYMBOL_TAG: {
422 // Find the null terminator
423 const std::byte* nullByte = std::find(data, data + remainingSize, std::byte{0});
424 if (nullByte == data + remainingSize) {
425 throw MalformedPacketException("String has no null terminator");
426 }
427
428 // Calculate string length and padded size
429 size_t strLen = nullByte - data;
430 size_t paddedSize = padSize(strLen + 1); // Include null terminator
431
432 if (remainingSize < paddedSize) {
433 throw BufferOverflowException("Not enough data for padded string");
434 }
435
436 // Create string from data
437 std::string str(reinterpret_cast<const char*>(data), strLen);
438
439 // Update data pointer and remaining size
440 data += paddedSize;
441 remainingSize -= paddedSize;
442
443 return (typeTag == STRING_TAG) ? Value(std::move(str))
444 : Value(Symbol(std::move(str)));
445 }
446 case BLOB_TAG: {
447 if (remainingSize < sizeof(Int32)) {
448 throw BufferOverflowException("Not enough data for blob size");
449 }
450
451 // Get blob size (big endian)
452 Int32 blobSize;
453 std::memcpy(&blobSize, data, sizeof(Int32));
454 blobSize = bigEndian(blobSize);
455
456 if (blobSize < 0) {
457 throw MalformedPacketException("Negative blob size");
458 }
459
460 if (static_cast<size_t>(blobSize) > MAX_BLOB_SIZE) {
461 throw MessageSizeException("Blob size too large");
462 }
463
464 data += sizeof(Int32);
465 remainingSize -= sizeof(Int32);
466
467 size_t paddedSize = padSize(static_cast<size_t>(blobSize));
468 if (remainingSize < paddedSize) {
469 throw BufferOverflowException("Not enough data for blob");
470 }
471
472 // Create blob from data
473 Blob blob(data, static_cast<size_t>(blobSize));
474
475 // Update data pointer and remaining size
476 data += paddedSize;
477 remainingSize -= paddedSize;
478
479 return Value(std::move(blob));
480 }
481 case TIMETAG_TAG: {
482 if (remainingSize < sizeof(uint64_t)) {
483 throw BufferOverflowException("Not enough data for timetag");
484 }
485
486 uint64_t ntp;
487 std::memcpy(&ntp, data, sizeof(uint64_t));
488 ntp = bigEndian(ntp);
489
490 data += sizeof(uint64_t);
491 remainingSize -= sizeof(uint64_t);
492
493 return Value(TimeTag(ntp));
494 }
495 case CHAR_TAG: {
496 if (remainingSize < sizeof(Int32)) {
497 throw BufferOverflowException("Not enough data for char");
498 }
499
500 Int32 val;
501 std::memcpy(&val, data, sizeof(Int32));
502 val = bigEndian(val);
503
504 data += sizeof(Int32);
505 remainingSize -= sizeof(Int32);
506
507 return Value(static_cast<Char>(val));
508 }
509 case RGBA_TAG: {
510 if (remainingSize < sizeof(uint32_t)) {
511 throw BufferOverflowException("Not enough data for RGBA");
512 }
513
514 uint32_t rgba;
515 std::memcpy(&rgba, data, sizeof(uint32_t));
516 rgba = bigEndian(rgba);
517
518 RGBAColor color(static_cast<uint8_t>((rgba >> 24) & 0xFF),
519 static_cast<uint8_t>((rgba >> 16) & 0xFF),
520 static_cast<uint8_t>((rgba >> 8) & 0xFF),
521 static_cast<uint8_t>(rgba & 0xFF));
522
523 data += sizeof(uint32_t);
524 remainingSize -= sizeof(uint32_t);
525
526 return Value(color);
527 }
528 case MIDI_TAG: {
529 if (remainingSize < 4) {
530 throw BufferOverflowException("Not enough data for MIDI");
531 }
532
533 MIDIMessage midi(static_cast<uint8_t>(std::to_integer<uint8_t>(data[0])),
534 static_cast<uint8_t>(std::to_integer<uint8_t>(data[1])),
535 static_cast<uint8_t>(std::to_integer<uint8_t>(data[2])),
536 static_cast<uint8_t>(std::to_integer<uint8_t>(data[3])));
537
538 data += 4;
539 remainingSize -= 4;
540
541 return Value(midi);
542 }
543 case TRUE_TAG:
544 return Value::trueBool();
545 case FALSE_TAG:
546 return Value::falseBool();
547 case NIL_TAG:
548 return Value::nil();
549 case INFINITUM_TAG:
550 return Value::infinitum();
551 default:
552 throw UnknownTypeException("Unknown type tag");
553 }
554 } catch (const OSCException&) {
555 throw;
556 } catch (const std::exception& e) {
557 throw DeserializationException("Error deserializing type '" + std::string(1, typeTag) +
558 "': " + std::string(e.what()));
559 }
