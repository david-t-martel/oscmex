61 AddressImpl::AddressImpl(std::string host, std::string port, Protocol protocol)
62 : host_(std::move(host)), port_(std::move(port)), protocol_(protocol) {
63 // Initialize networking subsystem if needed
64 if (!networkingInitialized) {
65 try {
66 initializeNetworking();
67 } catch (const OSCException &e) {
68 // Re-throw with improved context
69 throw OSCException("Failed to initialize networking for address " + host_ + ":" +
70 port_ + ": " + e.what(),
71 e.code());
72 }
73 }
74
75 // Initialize socket and resolve address
76 if (networkingInitialized) {
77 try {
78 if (!resolveAddress()) {
79 throw AddressException("Failed to resolve host '" + host_ + "' with port '" +
80 port_ + "': " + getSystemErrorMessage(),
81 OSCException::ErrorCode::AddressError);
82 }
83
84 if (!initializeSocket()) {
85 throw SocketException("Failed to initialize socket for " + host_ + ":" + port_ +
86 ": " + getSystemErrorMessage(),
87 OSCException::ErrorCode::SocketError);
88 }
89 } catch (const OSCException &e) {
90 // Clean up any resources that might have been allocated
91 cleanup();
92 // Re-throw the exception
93 throw;
94 } catch (const std::exception &e) {
95 // Clean up any resources that might have been allocated
96 cleanup();
97 // Wrap in our standard exception
98 throw AddressException("Address initialization error: " + std::string(e.what()),
99 OSCException::ErrorCode::AddressError);
100 }
101 }

120 // Initialize the socket
121 bool AddressImpl::initializeSocket() {
122 if (!addrInfo_) {
123 return false;
124 }
125
126 // Try each address until we successfully create a socket
127 struct addrinfo *addr;
128 for (addr = addrInfo_; addr != nullptr; addr = addr->ai_next) {
129 socket_ = socket(addr->ai_family, addr->ai_socktype, addr->ai_protocol);
130 if (socket_ != INVALID_SOCKET_VALUE) {
131 // For client sockets (which is what AddressImpl is designed for),
132 // we don't need to bind to a specific local address.
133 // This allows the OS to choose an appropriate port.
134 break;
135 }
136 }
137
138 if (socket_ == INVALID_SOCKET_VALUE) {
139 return false;
140 }
141
142 return true;
143 }

145 // Resolve the address
146 bool AddressImpl::resolveAddress() {
147 struct addrinfo hints;
148 std::memset(&hints, 0, sizeof(hints));
149
150 // Set appropriate address family and socket type
151 switch (protocol_) {
152 case Protocol::UDP:
153 hints.ai_family = AF_UNSPEC;
154 hints.ai_socktype = SOCK_DGRAM;
155 break;
156 case Protocol::TCP:
157 hints.ai_family = AF_UNSPEC;
158 hints.ai_socktype = SOCK_STREAM;
159 break;
160 case Protocol::UNIX:
161#ifdef _WIN32
162 // Windows doesn't support UNIX domain sockets in older versions
163 throw PlatformException("UNIX domain sockets not supported on Windows",
164 OSCException::ErrorCode::NotImplemented);
165#else
166 hints.ai_family = AF_UNIX;
167 hints.ai_socktype = SOCK_STREAM;
168 break;
169#endif
170 }
171
172 // Resolve the address
173 int result = getaddrinfo(host_.c_str(), port_.c_str(), &hints, &addrInfo_);
174 if (result != 0) {
175 std::string errorMsg;
176#ifdef _WIN32
177 errorMsg = gai_strerrorA(result);
178#else
179 errorMsg = gai_strerror(result);
180#endif
181 throw AddressException(
182 "Failed to resolve host '" + host_ + "' with port '" + port_ + "': " + errorMsg,
183 OSCException::ErrorCode::AddressError);
184 }
185
186 return true;
187 }

189 // Send data over the socket
190 bool AddressImpl::send(const std::vector<std::byte> &data) {
191 if (socket_ == INVALID_SOCKET_VALUE || !addrInfo_) {
192 throw SocketException("Cannot send OSC data: Socket not initialized",
193 OSCException::ErrorCode::SocketError);
194 }
195
196 if (data.size() > MAX_MESSAGE_SIZE) {
197 throw MessageSizeException("Message exceeds maximum allowed size (" +
198 std::to_string(data.size()) + " > " +
199 std::to_string(MAX_MESSAGE_SIZE) + " bytes)",
200 OSCException::ErrorCode::MessageTooLarge);
201 }
202
203 ssize_t bytesSent = 0;
204
205 try {
206 switch (protocol_) {
207 case Protocol::UDP:
208 // For UDP, we need to specify the destination each time
209 bytesSent = sendto(socket_, reinterpret_cast<const char *>(data.data()),
210 data.size(), 0, addrInfo_->ai_addr, addrInfo_->ai_addrlen);
211 break;
212
213 case Protocol::TCP:
214 if (!connected_) {
215 // Connect to the server
216 if (connect(socket_, addrInfo_->ai_addr, addrInfo_->ai_addrlen) ==
217 SOCKET_ERROR_VALUE) {
218 throw NetworkException("Failed to connect to " + host_ + ":" + port_ +
219 ": " + getSystemErrorMessage(),
220 OSCException::ErrorCode::NetworkError);
221 }
222 connected_ = true;
223 }
224
225 // For TCP, first send the size
226 uint32_t size = htonl(static_cast<uint32_t>(data.size()));
227 if (::send(socket_, reinterpret_cast<const char *>(&size), sizeof(size), 0) !=
228 sizeof(size)) {
229 throw SerializationException(
230 "Failed to send message size: " + getSystemErrorMessage(),
231 OSCException::ErrorCode::SerializationError);
232 }
233
234 // Then send the actual data
235 bytesSent = ::send(socket_, reinterpret_cast<const char *>(data.data()),
236 data.size(), 0);
237 break;
238
239 case Protocol::UNIX:
240#ifdef _WIN32
241 throw PlatformException("UNIX domain sockets not supported on this platform",
242 OSCException::ErrorCode::NotImplemented);
243#else
244 if (!connected_) {
245 // Connect to the server
246 if (connect(socket_, addrInfo_->ai_addr, addrInfo_->ai_addrlen) ==
247 SOCKET_ERROR_VALUE) {
248 throw NetworkException("Failed to connect to UNIX socket " + host_ +
249 ": " + getSystemErrorMessage(),
250 ErrorCode::NetworkError);
251 }
252 connected_ = true;
253 }
254 // Send the data
255 bytesSent = ::send(socket_, reinterpret_cast<const char *>(data.data()),
256 data.size(), 0);
257 break;
258#endif
259 }
260
261 if (bytesSent == SOCKET_ERROR_VALUE) {
262 throw NetworkException("Error sending OSC data: " + getSystemErrorMessage(),
263 OSCException::ErrorCode::NetworkError);
264 }
265
266 // Check if all data was sent
267 if (static_cast<size_t>(bytesSent) != data.size()) {
268 throw NetworkException("Incomplete OSC data transmission: Sent " +
269 std::to_string(bytesSent) + " of " +
270 std::to_string(data.size()) + " bytes",
271 OSCException::ErrorCode::NetworkError);
272 }
273
274 return true;
275 } catch (const OSCException &) {
276 // Re-throw OSC exceptions
277 throw;
278 } catch (const std::exception &e) {
279 // Wrap other exceptions
280 throw NetworkException("Error sending OSC data: " + std::string(e.what()),
281 OSCException::ErrorCode::NetworkError);
282 }
283 }

314 // Set the TTL for multicast
315 bool AddressImpl::setTTL(int ttl) {
316 if (socket_ == INVALID_SOCKET_VALUE || protocol_ != Protocol::UDP) {
317 throw InvalidArgumentException("Cannot set TTL: Invalid socket or protocol not UDP",
318 OSCException::ErrorCode::InvalidArgument);
319 }
320
321 // Clamp TTL to valid range
322 ttl = std::max(1, std::min(ttl, 255));
323
324 unsigned char ttlVal = static_cast<unsigned char>(ttl);
325 int result = setsockopt(socket_, IPPROTO_IP, IP_MULTICAST_TTL,
326 reinterpret_cast<const char *>(&ttlVal), sizeof(ttlVal));
327
328 if (result == SOCKET_ERROR_VALUE) {
329 throw NetworkException("Failed to set multicast TTL: " + getSystemErrorMessage(),
330 OSCException::ErrorCode::NetworkError);
331 }
332
333 ttl_ = ttl;
334 return true;
335 }

357 // Set socket timeout
358 bool AddressImpl::setTimeout(std::chrono::milliseconds timeout) {
359 if (socket_ == INVALID_SOCKET_VALUE) {
360 throw SocketException("Cannot set socket timeout: Invalid socket",
361 OSCException::ErrorCode::SocketError);
362 }
363
364 struct timeval tv;
365 tv.tv_sec = static_cast<long>(timeout.count() / 1000);
366 tv.tv_usec = static_cast<long>((timeout.count() % 1000) * 1000);
367
368 int result = setsockopt(socket_, SOL_SOCKET, SO_RCVTIMEO,
369 reinterpret_cast<const char *>(&tv), sizeof(tv));
370
371 if (result == SOCKET_ERROR_VALUE) {
372 throw NetworkException("Failed to set socket timeout: " + getSystemErrorMessage(),
373 OSCException::ErrorCode::NetworkError);
374 }
375
376 return true;
377 }
