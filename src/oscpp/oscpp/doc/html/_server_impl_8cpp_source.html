174 // Receive and dispatch a message
175 bool ServerImpl::receive(std::chrono::milliseconds timeout) {
176 if (socket_ == INVALID_SOCKET_VALUE) {
177 return false;
178 }
179
180 // Wait for data if timeout is specified and > 0
181 if (timeout.count() > 0 && !wait(timeout)) {
182 return false;
183 }
184
185 // Buffer to hold the incoming message
186 std::vector<std::byte> buffer(maxMessageSize_);
187
188 // Receive data
189 ssize_t bytesReceived = 0;
190 sockaddr_storage senderAddr;
191 socklen_t senderAddrLen = sizeof(senderAddr);
192
193 switch (protocol_) {
194 case Protocol::UDP:
195 // For UDP, use recvfrom to get sender address
196 bytesReceived =
197 recvfrom(socket_, reinterpret_cast<char *>(buffer.data()), buffer.size(), 0,
198 reinterpret_cast<sockaddr *>(&senderAddr), &senderAddrLen);
199 break;
200
201 case Protocol::TCP:
202 case Protocol::UNIX:
203 // Use framing for stream protocols
204 if (!tcp_framing::receiveFramed(socket_, buffer, maxMessageSize_)) {
205 if (errorHandler_) {
206 errorHandler_(getLastSocketError(), "Error receiving TCP message",
207 "ServerImpl::receive");
208 }
209 return false;
210 }
211 bytesReceived = buffer.size();
212 break;
213 }
214
215 if (bytesReceived <= 0) {
216 return false;
217 }
218
219 // Resize buffer to actual received size
220 buffer.resize(bytesReceived);
221
222 // Check if this is a bundle or a message
223 if (bytesReceived >= 8 && std::memcmp(buffer.data(), "#bundle", 7) == 0) {
224 // Handle as bundle
225 try {
226 Bundle bundle = Bundle::deserialize(buffer.data(), buffer.size());
227 dispatchBundle(bundle);
228 return true;
229 } catch (const OSCException &e) {
230 if (errorHandler_) {
231 errorHandler_(static_cast<int>(e.code()), e.what(), "ServerImpl::receive");
232 }
233 return false;
234 }
235 } else {
236 // Handle as message
237 try {
238 Message message = Message::deserialize(buffer.data(), buffer.size());
239 dispatchMessage(message);
240 return true;
241 } catch (const OSCException &e) {
242 if (errorHandler_) {
243 errorHandler_(static_cast<int>(e.code()), e.what(), "ServerImpl::receive");
244 }
245 return false;
246 }
247 }

440 // Dispatch a message to registered handlers
441 void ServerImpl::dispatchMessage(const Message &message) {
442 std::lock_guard<std::mutex> lock(methodMutex_);
443
444 // Get the path and build type tag string from arguments
445 std::string path = message.getPath();
446 std::string types;
447 for (const auto &arg : message.getArguments()) {
448 types += arg.typeTag();
449 }
450
451 // Find matching methods
452 std::vector<Method *> matchedMethods;
453 bool found = matchMethod(path, types, matchedMethods);
454
455 // Call matched methods
456 for (auto *method : matchedMethods) {
457 if (method && method->handler) {
458 try {
459 method->handler(message);
460 } catch (const std::exception &e) {
461 if (errorHandler_) {
462 errorHandler_(0, std::string("Exception in method handler: ") + e.what(),
463 "ServerImpl::dispatchMessage");
464 }
465 }
466 }
467 }
468
469 // If no methods matched and we have a default handler, call it
470 if (!found) {
471 for (auto &pair : methods_) {
472 if (pair.second.isDefault && pair.second.handler) {
473 try {
474 pair.second.handler(message);
475 } catch (const std::exception &e) {
476 if (errorHandler_) {
477 errorHandler_(0,
478 std::string("Exception in default handler: ") + e.what(),
479 "ServerImpl::dispatchMessage");
480 }
481 }
482 break; // Only call the first default handler
483 }
484 }
485 }

520 // Convert OSC pattern to regex
521 std::regex ServerImpl::convertPathToRegex(const std::string &pattern) {
522 std::string regexStr;
523 regexStr.reserve(pattern.size() * 2); // Reserve space to avoid reallocations
524 regexStr += "^"; // Match start of string
525
526 for (size_t i = 0; i < pattern.size(); ++i) {
527 char c = pattern[i];
528 switch (c) {
529 case '.': // Escape special regex chars
530 case '^':
531 case '$':
532 case '+':
533 case '\\':
534 regexStr += '\\';
535 regexStr += c;
536 break;
537 case '?': // OSC '?' matches any single character except '/'
538 regexStr += "[^/]";
539 break;
540 case '*': // OSC '*' matches any sequence of zero or more characters except '/'
541 regexStr += "[^/]*";
542 break;
543 case '[': // OSC '[abc]' matches any character in the set
544 {
545 regexStr += '[';
546 size_t closePos = pattern.find(']', i);
547 if (closePos == std::string::npos) {
548 throw OSCException("Unclosed '[' in pattern",
549 OSCException::ErrorCode::PatternError);
550 }
551 // Copy the bracket contents verbatim
552 regexStr += pattern.substr(i + 1, closePos - i - 1);
553 regexStr += ']';
554 i = closePos; // Skip past the matched bracket
555 break;
556 }
557 case '{': // OSC '{foo,bar}' matches any of the comma-separated strings
558 {
559 regexStr += '(';
560 size_t closePos = pattern.find('}', i);
561 if (closePos == std::string::npos) {
562 throw OSCException("Unclosed '{' in pattern",
563 OSCException::ErrorCode::PatternError);
564 }
565 // Replace commas with '|'
566 std::string options = pattern.substr(i + 1, closePos - i - 1);
567 for (size_t j = 0; j < options.size(); ++j) {
568 if (options[j] == ',') {
569 regexStr += '|';
570 } else {
571 regexStr += options[j];
572 }
573 }
574 regexStr += ')';
575 i = closePos; // Skip past the matched brace
576 break;
577 }
578 case '/': // Forward slash is a literal in OSC patterns
579 regexStr += '/';
580 break;
581 default:
582 regexStr += c;
583 break;
584 }
585 }
586
587 regexStr += "$"; // Match end of string
588 return std::regex(regexStr);
589 }

591 // Match a message against registered methods
592 bool ServerImpl::matchMethod(const std::string &path, const std::string &types,
593 std::vector<Method *> &matchedMethods) {
594 bool found = false;
595
596 try {
597 for (auto &pair : methods_) {
598 Method &method = pair.second;
599
600 // Skip default methods, they're handled separately
601 if (method.isDefault) {
602 continue;
603 }
604
605 // Try to match the path using our direct pattern matching function
606 if (matchPattern(method.pathPattern, path)) {
607 // Path matches, now check the type spec
608 if (method.typeSpec.empty() ||
609 (types.length() >= method.typeSpec.length() &&
610 types.substr(0, method.typeSpec.length()) == method.typeSpec)) {
611 // Both path and types match
612 matchedMethods.push_back(&method);
613 found = true;
614 } else if (errorHandler_) {
615 // Path matched but types didn't - this is useful debug information
616 errorHandler_(static_cast<int>(OSCException::ErrorCode::TypeMismatch),
617 "Type signature mismatch for " + path + ": expected '" +
618 method.typeSpec + "' but got '" + types + "'",
619 "ServerImpl::matchMethod");
620 }
621 }
622 }
623 } catch (const std::exception &e) {
624 if (errorHandler_) {
625 errorHandler_(static_cast<int>(OSCException::ErrorCode::MatchError),
626 "Error matching OSC method: " + std::string(e.what()),
627 "ServerImpl::matchMethod");
628 }
629 }
630
631 return found;
632 }

636 // Send data with proper size framing
637 bool sendFramed(SOCKET_TYPE socket, const std::vector<std::byte> &data) {
638 // Prepare size in network byte order (big-endian)
639 uint32_t size = static_cast<uint32_t>(data.size());
640 uint32_t sizeBE = htonl(size);
641
642 // Send the size
643 ssize_t sizeResult =
644 send(socket, reinterpret_cast<const char *>(&sizeBE), sizeof(sizeBE), 0);
645
646 if (sizeResult != sizeof(sizeBE)) {
647 int errorCode =
648#ifdef _WIN32
649 WSAGetLastError();
650#else
651 errno;
652#endif
653 throw OSCException("Failed to send TCP frame size: " + std::to_string(errorCode),
654 OSCException::ErrorCode::SerializationError);
655 return false;
656 }
657
658 // Send the data
659 ssize_t dataResult =
660 send(socket, reinterpret_cast<const char *>(data.data()), data.size(), 0);
661
662 if (dataResult != static_cast<ssize_t>(data.size())) {
663 int errorCode =
664#ifdef _WIN32
665 WSAGetLastError();
666#else
667 errno;
668#endif
669 throw OSCException("Failed to send TCP frame data: Sent " +
670 std::to_string(dataResult) + " of " +
671 std::to_string(data.size()) + " bytes",
672 OSCException::ErrorCode::SerializationError);
673 return false;
674 }
675
676 return true;
677 }

679 // Receive data with size framing
680 bool receiveFramed(SOCKET_TYPE socket, std::vector<std::byte> &buffer, size_t maxSize) {
681 try {
682 // First read the 4-byte size
683 uint32_t sizeBE = 0;
684 ssize_t bytesReceived =
685 recv(socket, reinterpret_cast<char *>(&sizeBE), sizeof(sizeBE), 0);
686
687 if (bytesReceived != sizeof(sizeBE)) {
688 // Could be due to a closed connection or other error
689 if (bytesReceived == 0) {
690 // Connection closed by peer
691 return false;
692 } else if (bytesReceived < 0) {
693 int errorCode =
694#ifdef _WIN32
695 WSAGetLastError();
696#else
697 errno;
698#endif
699 throw OSCException(
700 "Error receiving TCP frame size: " + std::to_string(errorCode),
701 OSCException::ErrorCode::DeserializationError);
702 } else {
703 throw OSCException(
704 "Incomplete TCP frame size received: " + std::to_string(bytesReceived) +
705 " of " + std::to_string(sizeof(sizeBE)) + " bytes",
706 OSCException::ErrorCode::DeserializationError);
707 }
708 return false;
709 }
710
711 // Convert from network byte order
712 uint32_t size = ntohl(sizeBE);
713
714 // Check against maximum allowed size
715 if (size > maxSize) {
716 throw OSCException("TCP message size exceeds maximum allowed (" +
717 std::to_string(size) + " > " + std::to_string(maxSize) +
718 " bytes)",
719 OSCException::ErrorCode::MessageTooLarge);
720 }
721
722 // Check for suspiciously small size that might be a framing error
723 if (size < 8) { // Minimum valid OSC packet should be at least 8 bytes
724 throw OSCException(
725 "TCP message size suspiciously small: " + std::to_string(size) + " bytes",
726 OSCException::ErrorCode::DeserializationError);
727 }
728
729 // Resize buffer to fit the data
730 buffer.resize(size);
731
732 // Read the data in chunks (safer for large messages)
733 size_t totalReceived = 0;
734 int retryCount = 0;
735 const int MAX_RETRIES = 5;
736
737 while (totalReceived < size) {
738 bytesReceived =
739 recv(socket, reinterpret_cast<char *>(buffer.data() + totalReceived),
740 size - totalReceived, 0);
741
742 if (bytesReceived < 0) {
743 // Check if it's a retriable error (e.g., interrupted system call)
744#ifdef _WIN32
745 int errorCode = WSAGetLastError();
746 if (errorCode == WSAEINTR && retryCount < MAX_RETRIES) {
747#else
748 int errorCode = errno;
749 if (errorCode == EINTR && retryCount < MAX_RETRIES) {
750#endif
751 retryCount++;
752 continue; // Retry the receive
753 }
754
755 // Non-retriable error
756 throw OSCException(
757 "Error receiving TCP frame data: " + std::to_string(errorCode),
758 OSCException::ErrorCode::DeserializationError);
759 return false;
760 } else if (bytesReceived == 0) {
761 // Connection closed by peer in the middle of a frame
762 throw OSCException("Connection closed during frame reception after " +
763 std::to_string(totalReceived) + " of " +
764 std::to_string(size) + " bytes",
765 OSCException::ErrorCode::DeserializationError);
766 return false;
767 }
768
769 totalReceived += bytesReceived;
770 }
771
772 return true;
773 } catch (const OSCException &) {
774 // Allow OSCExceptions to propagate
775 throw;
776 } catch (const std::exception &e) {
777 // Wrap other exceptions
778 throw OSCException("Error in TCP frame handling: " + std::string(e.what()),
779 OSCException::ErrorCode::DeserializationError);
780 }
781 }

784 // OSC pattern matching function that directly implements the OSC spec
785 // without relying on regex for better performance and correctness
786 bool ServerImpl::matchPattern(const std::string &pattern, const std::string &path) {
787 size_t p = 0; // Pattern index
788 size_t s = 0; // String (path) index
789
790 while (p < pattern.size() && s < path.size()) {
791 char patternChar = pattern[p];
792
793 switch (patternChar) {
794 case '?': // Match any single character except '/'
795 if (path[s] == '/') {
796 return false;
797 }
798 p++;
799 s++;
800 break;
801
802 case '*': // Match any sequence of characters except '/'
803 {
804 // Skip consecutive stars (they're redundant)
805 while (p + 1 < pattern.size() && pattern[p + 1] == '*') {
806 p++;
807 }
808
809 // If star is at the end, it matches everything except '/'
810 if (p + 1 >= pattern.size()) {
811 // Ensure remaining path doesn't contain any '/'
812 while (s < path.size()) {
813 if (path[s] == '/') {
814 return false;
815 }
816 s++;
817 }
818 return true;
819 }
820
821 // Try different match lengths for the star
822 for (size_t i = 0; i <= path.size() - s; i++) {
823 // Ensure the * doesn't match a '/'
824 bool containsSlash = false;
825 for (size_t j = 0; j < i; j++) {
826 if (path[s + j] == '/') {
827 containsSlash = true;
828 break;
829 }
830 }
831
832 if (containsSlash) {
833 break;
834 }
835
836 // Try to match the rest of the pattern starting from here
837 if (matchPattern(pattern.substr(p + 1), path.substr(s + i))) {
838 return true;
839 }
840 }
841 return false;
842 }
843
844 case '[': // Character class
845 {
846 p++; // Skip the '['
847
848 // Check for negation
849 bool negate = false;
850 if (p < pattern.size() && pattern[p] == '!') {
851 negate = true;
852 p++;
853 }
854
855 bool matched = false;
856 size_t startPos = p;
857
858 // Find closing bracket
859 while (p < pattern.size() && pattern[p] != ']') {
860 if (p + 2 < pattern.size() && pattern[p + 1] == '-') {
861 // Range match
862 char startChar = pattern[p];
863 char endChar = pattern[p + 2];
864
865 if (path[s] >= startChar && path[s] <= endChar) {
866 matched = true;
867 }
868
869 p += 3;
870 } else {
871 // Direct character match
872 if (pattern[p] == path[s]) {
873 matched = true;
874 }
875 p++;
876 }
877 }
878
879 if (p >= pattern.size()) {
880 // Unterminated bracket
881 return false;
882 }
883
884 p++; // Skip closing bracket
885
886 if (negate) {
887 matched = !matched;
888 }
889
890 if (!matched) {
891 return false;
892 }
893
894 s++; // Move past the matched character
895 break;
896 }
897
898 case '{': // Alternatives
899 {
900 p++; // Skip the '{'
901
902 size_t closeBrace = pattern.find('}', p);
903 if (closeBrace == std::string::npos) {
904 // Unterminated brace
905 return false;
906 }
907
908 // Split by commas and try each alternative
909 std::string alternatives = pattern.substr(p, closeBrace - p);
910 std::string remaining = pattern.substr(closeBrace + 1);
911
912 size_t start = 0;
913 size_t comma;
914
915 while ((comma = alternatives.find(',', start)) != std::string::npos) {
916 std::string option = alternatives.substr(start, comma - start);
917
918 if (matchPattern(option + remaining, path.substr(s))) {
919 return true;
920 }
921
922 start = comma + 1;
923 }
924
925 // Try the last alternative
926 std::string lastOption = alternatives.substr(start);
927 return matchPattern(lastOption + remaining, path.substr(s));
928 }
929
930 default:
931 // Regular character match
932 if (patternChar != path[s]) {
933 return false;
934 }
935 p++;
936 s++;
937 break;
938 }
939 }
940
941 // Skip trailing stars
942 while (p < pattern.size() && pattern[p] == '*') {
943 p++;
944 }
945
946 // Both pattern and path should be fully consumed
947 return (p == pattern.size() && s == path.size());
948 }
